def clamp(value, min_value, max_value):

[docs]    '''Restricts a value to be within a specified range'''
    return max(min_value, min(value, max_value))

def lerp(value1, value2, amount):


[docs]    '''Linearly interpolates between two values'''
    return value1 + (value2-value1)*amount

def smoothstep(value1, value2, amount):


[docs]    '''Interpolates between two values using a cubic equation'''
    v = (amount*amount*(3 - 2*amount))
    return (value2 * v) + (value1 * (1 - v))

def hermite(p0, r0, p1, r1, t):


[docs]    '''Performs a Hermite spline interpolation'''
    t2 = t * t
    t3 = t2 * t
    _3t2 = 3.0 * t2
    _2t3 = 2.0 * t3

    return (p0*(_2t3-_3t2+1) + p1*(-_2t3+_3t2) + r0*(t3-2.0*t2+t) + r1*(t3-t2))

def catmullrom(p_1, p0, p1, p2, t):


[docs]    '''Performs a Catmull-Rom interpolation using the specified positions'''
    return (
          t*((2-t)*t - 1)   * p_1
        + (t*t*(3*t - 5) + 2) * p0
        + t*((4 - 3*t)*t + 1) * p1
        + (t-1)*t*t         * p2 ) / 2.0

def wrap_angle(angle):


[docs]    '''Reduces a given angle to a value between pi and -pi'''
    if angle > 0:
        result = fmod(angle+pi, two_pi) - pi;
    else:
        result = fmod(angle-pi, two_pi) + pi;

    return result

def distance(value1, value2):


[docs]    '''Calculates the absolute value of the difference of two values'''
    if not is_iterable(value1): value1 = [value1]
    if not is_iterable(value2): value2 = [value2]

    return sqrt(sum((v1-v2)**2 for v1, v2 in zip(value1, value2)))
