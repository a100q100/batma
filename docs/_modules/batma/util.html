    '''Return a list of floats'''
    return [x for x in xfrange(start, stop, step)]



[docs]def xfrange(start, stop=None, step=1):
    '''Return a generator of floats'''
    step = float(step)
    if stop is None:
        start, stop = 0, start

    i = start
    while i < stop:
        yield i
        i += step




[docs]def is_iterable(value):
    '''Verify if a ``value`` is an iterable'''
    try:
        iter(value)
    except TypeError:
        return False

    return True



[docs]def singleton(cls):
    '''Singleton decorator'''
    instances = {}
    def getinstance(*args, **kwargs):
        if cls not in instances:
            instances[cls] = cls(*args, **kwargs)
        return instances[cls]
    return getinstance
